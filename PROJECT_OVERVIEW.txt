      cpu.setCPSR(0xF0000013);  // Set flags + Supervisor mode

   5. Program execution with memory
      cpu.setMemoryCallbacks(readMem, writeMem);
      cpu.step();  // Fetch → Decode → Execute

      std::cout << cpu.getStateString();
